% Executive contact header

        \textcolor{charcoal}{\small
            #2 \quad \textcolor{gold}{|} \quad

            \href{mailto:#4}{#4}
        } \\[4pt]

            \href{#5}{\faIcon{linkedin} LinkedIn} \quad \textcolor{gold}{|} \quad

}
