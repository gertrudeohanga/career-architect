% --- Creative Modern Style ---
% A bold, eye-catching resume template with color accents
% Best for: Startups, creative roles, design-forward companies

\usepackage[empty]{fullpage}
\usepackage{titlesec}
\usepackage{xcolor}
\usepackage[unicode]{hyperref}
\usepackage{enumitem}
\usepackage{fontawesome5}
\usepackage{setspace}
\usepackage{parskip}
\usepackage{tikz}

% Bold color palette
\definecolor{primary}{HTML}{6366F1}    % Indigo
\definecolor{secondary}{HTML}{EC4899}  % Pink
\definecolor{accent}{HTML}{14B8A6}     % Teal
\definecolor{dark}{HTML}{1F2937}       % Dark gray

% Generous margins for breathing room
\addtolength{\oddsidemargin}{-0.3in}
\addtolength{\evensidemargin}{-0.3in}
\addtolength{\textwidth}{0.6in}
\addtolength{\topmargin}{-.3in}
\addtolength{\textheight}{0.6in}

% Bold section headers with color bar
\titleformat{\section}{
    \vspace{-2pt}\Large\bfseries\color{primary}
}{}{0em}{}[\color{primary}\titlerule[2pt]\vspace{-2pt}]

% Colorful links
\hypersetup{colorlinks=true, urlcolor=primary, linkcolor=primary}

% Creative contact header with icons
\newcommand{\contactline}[7]{
    \begin{center}
        {\fontsize{28}{34}\selectfont\textbf{\textcolor{dark}{#1}}} \\[10pt]
        \textcolor{primary}{\faIcon{map-marker-alt}} #2 \quad
        \textcolor{secondary}{\faIcon{phone}} #3 \quad
        \textcolor{accent}{\faIcon{envelope}} \href{mailto:#4}{#4} \\[6pt]
        \textcolor{primary}{\faIcon{linkedin}} \href{#5}{LinkedIn} \quad
        \textcolor{secondary}{\faIcon{github}} \href{#6}{GitHub} \quad
        \textcolor{accent}{\faIcon{globe}} \href{#7}{Portfolio}
    \end{center}
    \vspace{14pt}
}

% Stylish bullet points
\setlist[itemize]{leftmargin=0.15in, labelsep=0.5em, nosep, itemsep=4pt}
\renewcommand{\labelitemi}{\textcolor{primary}{\faIcon{chevron-right}}}

% Numbered list styling with color
\setlist[enumerate]{leftmargin=0.2in, labelsep=0.5em, itemsep=4pt, topsep=4pt, parsep=2pt}
\setlist[enumerate,1]{label=\textcolor{primary}{\arabic*.}}
\setlist[enumerate,2]{label=\textcolor{secondary}{\alph*.}, leftmargin=0.3in}
\setlength{\parskip}{5pt}
\onehalfspacing

\pagenumbering{gobble}
